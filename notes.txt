Endpoints:

Likes

{
    Like: http://localhost:4000/api/posts/(post id)/like
    Remove Like: http://localhost:4000/api/posts/(post id)/removelike

    Dislike: http://localhost:4000/api/posts/(post id)/dislike
    Remove Dislike: http://localhost:4000/api/posts/(post id)/removedislike
}

Comments:

{
    Comment: http://localhost:4000/api/comments/(post id)
    Delete Comment: http://localhost:4000/api/comments/(post id)

    Like Comment: http://localhost:4000/api/comment/(post id)/like
    Remove Like Comment: http://localhost:4000/api/comment/(post id)/removelike

    Dislike Comment: http://localhost:4000/api/comment/(post id)/dislike
    Remove Dislike Comment: http://localhost:4000/api/comment/(post id)/removedislike

    New Comment Response: http://localhost:4000/api/responses/(post id)
    Delete Comment Response: http://localhost:4000/api/comments/(post id)

    Like Comment Response: http://localhost:3000/api/response/(post id)/like
    Delete Like Comment Response: http://localhost:3000/api/response/(post id)/removelike
}
